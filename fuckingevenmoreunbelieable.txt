import html
import re

def check_html_file(filepath):
    try:
        with open(filepath, 'r', encoding='utf-8') as f:
            content = f.read()

        # Basic checks
        if not content.strip().startswith('<!DOCTYPE html>'):
            print(f'{filepath}: Missing or incorrect DOCTYPE')
            return False

        # Check for unclosed tags (basic)
        open_tags = []
        for match in re.finditer(r'<(/?)(\w+)[^>]*>', content):
            is_closing = match.group(1) == '/'
            tag = match.group(2).lower()

            if tag in ['br', 'img', 'input', 'meta', 'link', 'hr', 'area', 'base', 'col', 'embed', 'param', 'source', 'track', 'wbr']:
                continue  # Self-closing tags

            if not is_closing:
                open_tags.append(tag)
            elif open_tags and open_tags[-1] == tag:
                open_tags.pop()
            else:
                print(f'{filepath}: Unmatched closing tag </{tag}>')
                return False

        if open_tags:
            print(f'{filepath}: Unclosed tags: {open_tags}')
            return False

        print(f'{filepath}: HTML structure looks good')
        return True

    except Exception as e:
        print(f'{filepath}: Error checking file: {e}')
        return False
